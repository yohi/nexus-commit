## 実行前提

### Devcontainer の利用 (CRITICAL)

すべてのテスト・静的解析・ビルドコマンドは **必ず Devcontainer 内で実行** してください。ホスト側 Node.js でコマンドを直接実行してはいけません。

Devcontainer に入る方法（いずれか 1 つを選ぶ）:

```bash
# 方法 A: VS Code / Cursor の Reopen in Container でコンテナ内ターミナルを開く
# 方法 B: @devcontainers/cli を使う（ホスト側）
npx @devcontainers/cli up --workspace-folder .
npx @devcontainers/cli exec --workspace-folder . -- bash
```

本計画書では、コマンドブロックに `# Devcontainer 内で実行` のコメントを付与します。これは「Devcontainer の bash 上で実行」を意味します。

### Worktree

実装は `.worktrees/` 配下に Phase 単位の worktree を作成して進めます (`.worktrees/` は `.gitignore` 済み、検証済み)。各 Phase 開始時に `superpowers:using-git-worktrees` スキル相当の手順で worktree を作成します。

### Git 運用ルール（再掲）

- **Phase ブランチ**: `master` から `feature/phaseX_<機能名>__base` を作成し、`master` 向け Draft PR を作成する。前 Phase の PR が `master` にマージされるまで、次 Phase の作業を始めない。
- **Task ブランチ**: 親は **Phase Base** または **直前の Task ブランチ**。
  - **独立タスク**（前 Task の未マージコードに依存しない）→ Phase Base から派生
  - **依存タスク**（前 Task の未マージコードを物理的に必要とする）→ 直前 Task ブランチから派生（`git checkout <prev-task> && git checkout -b <new-task>`）
- 各 Task 完了時に **Phase Base 向け Draft PR** を作成する。

```typescript
import { z } from 'zod';

/** Nexus API: POST /api/search レスポンス */
export const NexusResultItemSchema = z.object({
  file: z.string(),
  content: z.string(),
});

export const NexusSearchResponseSchema = z.object({
  results: z.array(NexusResultItemSchema),
});

/** OpenAI 互換: POST /v1/chat/completions レスポンス */
export const ChatCompletionResponseSchema = z
  .object({
    choices: z
      .array(
        z
          .object({
            message: z.object({ content: z.string() }).passthrough(),
          })
          .passthrough(),
      )
      .min(1, 'choices must contain at least one item'),
  })
  .passthrough();

/** OpenAI 互換: GET /v1/models レスポンス */
export const ModelListResponseSchema = z
  .object({
    data: z.array(z.object({ id: z.string() }).passthrough()),
  })
  .passthrough();

/** safeParse 失敗時のエラーメッセージ整形 */
export function formatZodError(prefix: string, err: z.ZodError): Error {
  const first = err.issues[0];
  if (first === undefined) {
    return new Error(`${prefix}: unknown validation error`);
  }
  const path = first.path.length > 0 ? ` at ${first.path.join('.')}` : '';
  return new Error(`${prefix}${path}: ${first.message}`);
}
```

```typescript
import { NexusSearchResponseSchema, formatZodError } from './schemas.js';
import type { NexusClientPort, NexusResult, NexusSearchRequest } from './types.js';

function parseResults(data: unknown): NexusResult[] {
  const parsed = NexusSearchResponseSchema.safeParse(data);
  if (!parsed.success) {
    throw formatZodError('Invalid Nexus response', parsed.error);
  }
  return parsed.data.results;
}

export class HttpNexusClient implements NexusClientPort {
  private readonly normalizedBaseUrl: string;
  constructor(baseUrl: string) {
    this.normalizedBaseUrl = baseUrl.replace(/\/+$/, '');
  }

  async search(req: NexusSearchRequest, opts: { timeoutMs: number }): Promise<NexusResult[]> {
    const timeout = Number.isFinite(opts.timeoutMs) && opts.timeoutMs > 0 ? opts.timeoutMs : 5000;

    const controller = new AbortController();
    const timer = setTimeout(() => {
      controller.abort();
    }, timeout);
    try {
      const res = await fetch(`${this.normalizedBaseUrl}/api/search`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(req),
        signal: controller.signal,
      });

      if (!res.ok) {
        throw new Error(`Nexus API error: ${res.status}`);
      }

      const data = (await res.json()) as unknown;
      return parseResults(data);
    } catch (err) {
      if (err instanceof Error && err.name === 'AbortError') {
        throw new Error('Nexus search timed out');
      }
      throw err;
    } finally {
      clearTimeout(timer);
    }
  }
}
```

```typescript
async listModels(opts: { timeoutMs: number }): Promise<string[]> {
  if (
    typeof opts.timeoutMs !== 'number' ||
    !Number.isFinite(opts.timeoutMs) ||
    opts.timeoutMs <= 0
  ) {
    throw new Error(`Invalid timeoutMs: ${opts.timeoutMs}. Must be a positive finite number.`);
  }
  const controller = new AbortController();
  const timer = setTimeout(() => {
    controller.abort();
  }, opts.timeoutMs);
  let res: Response;
  try {
    const url = new URL(
      'models',
      this.baseUrl.endsWith('/') ? this.baseUrl : `${this.baseUrl}/`,
    ).toString();
    res = await fetch(url, {
      method: 'GET',
      headers: {
        Authorization: `Bearer ${this.apiKey}`,
      },
      signal: controller.signal,
    });
  } catch (err) {
    if (err instanceof Error && err.name === 'AbortError') {
      throw new Error(`LLM models request timed out after ${opts.timeoutMs}ms`);
    }
    throw err;
  } finally {
    clearTimeout(timer);
  }
  if (!res.ok) {
    const body = await res.text().catch(() => '(failed to read body)');
    throw new Error(`LLM API error: ${res.status} ${res.statusText}\nBody: ${body}`);
  }
  const data = (await res.json()) as unknown;
  return extractModelIds(data);
}
```

```typescript
import type { Config, NexusClientPort, LlmClientPort } from './types.js';

export type CheckStatus = 'ok' | 'warn' | 'fail' | 'skip';

export interface CheckResult {
  readonly title: string;
  readonly status: CheckStatus;
  readonly detail?: string;
  readonly hint?: string;
}

export interface DoctorDeps {
  readonly nexus: NexusClientPort;
  readonly llm: LlmClientPort;
  readonly cwd?: string;
}

export interface DoctorReport {
  readonly results: readonly CheckResult[];
  readonly exitCode: 0 | 4;
}

const PROBE_TIMEOUT_CAP_MS = 3000;

function probeTimeout(base: number): number {
  return Math.min(base, PROBE_TIMEOUT_CAP_MS);
}

function checkNodeVersion(): CheckResult {
  const major = Number.parseInt(process.versions.node.split('.')[0] ?? '0', 10);
  if (major >= 22) {
    return { title: 'Node.js version', status: 'ok', detail: `v${process.versions.node}` };
  }
  return {
    title: 'Node.js version',
    status: 'fail',
    detail: `v${process.versions.node}`,
    hint: 'Node.js 22+ required. Use nvm to install: nvm install 22',
  };
}

function checkConfiguration(config: Config): CheckResult {
  const detail =
    `llmUrl=${config.llmUrl}, model=${config.llmModel}, ` +
    `lang=${config.lang}, maxChars=${config.maxChars}, apiKey=***`;
  return { title: 'Configuration', status: 'ok', detail };
}

async function checkNexus(config: Config, deps: DoctorDeps): Promise<CheckResult> {
  if (!config.useContext) {
    return {
      title: 'Nexus API reachable',
      status: 'skip',
      detail: 'useContext=false',
    };
  }
  const start = Date.now();
  try {
    await deps.nexus.search(
      { query: '', files: [] },
      { timeoutMs: probeTimeout(config.nexusTimeoutMs) },
    );
    return {
      title: 'Nexus API reachable',
      status: 'ok',
      detail: `${config.nexusUrl} (responded in ${Date.now() - start}ms)`,
    };
  } catch (err) {
    return {
      title: 'Nexus API reachable',
      status: 'fail',
      detail: err instanceof Error ? err.message : String(err),
      hint: `Nexus API not reachable at ${config.nexusUrl}. Is the server running?`,
    };
  }
}

async function checkLlmEndpoint(
  config: Config,
  deps: DoctorDeps,
): Promise<{ result: CheckResult; models: string[] | null }> {
  try {
    const models = await deps.llm.listModels({ timeoutMs: probeTimeout(config.llmTimeoutMs) });
    return {
      result: {
        title: 'LLM endpoint reachable',
        status: 'ok',
        detail: config.llmUrl,
      },
      models,
    };
  } catch (err) {
    return {
      result: {
        title: 'LLM endpoint reachable',
        status: 'fail',
        detail: err instanceof Error ? err.message : String(err),
        hint: `LLM endpoint not reachable at ${config.llmUrl}. Ensure your local LLM server is running.`,
      },
      models: null,
    };
  }
}

function checkModelExists(config: Config, models: string[] | null): CheckResult {
  const title = `Model '${config.llmModel}'`;
  if (models === null) {
    return { title, status: 'skip', detail: 'LLM endpoint check failed' };
  }
  if (models.includes(config.llmModel)) {
    return { title, status: 'ok' };
  }
  return {
    title,
    status: 'fail',
    detail: `Available models: ${models.join(', ') || '(empty)'}`,
    hint: `Run: ollama pull ${config.llmModel}`,
  };
}

async function checkPromptFile(_deps: DoctorDeps): Promise<CheckResult> {
  // Phase 2 では .github/nxc.prompt.md の検出は未実装。Phase 4 で findPromptFile を導入する際に実装される。
  return {
    title: 'Custom prompt file',
    status: 'skip',
    detail: 'not yet implemented',
  };
}

export async function runDoctor(config: Config, deps: DoctorDeps): Promise<DoctorReport> {
  const results: CheckResult[] = [];
  results.push(checkNodeVersion());
  results.push(checkConfiguration(config));
  results.push(await checkNexus(config, deps));
  const { result: llmResult, models } = await checkLlmEndpoint(config, deps);
  results.push(llmResult);
  results.push(checkModelExists(config, models));
  results.push(await checkPromptFile(deps));

  const exitCode: 0 | 4 = results.some((r) => r.status === 'fail') ? 4 : 0;
  return { results, exitCode };
}

const STATUS_GLYPH: Record<CheckStatus, string> = {
  ok: '✓',
  warn: '!',
  fail: '✗',
  skip: '⊘',
};

export function renderReport(report: DoctorReport): string {
  const lines: string[] = ['nxc doctor', ''];
  for (const r of report.results) {
    const glyph = STATUS_GLYPH[r.status];
    const detail = r.detail !== undefined ? `   ${r.detail}` : '';
    lines.push(`  ${glyph} ${r.title}${detail ? `\n      ${r.detail ?? ''}` : ''}`);
    if (r.hint !== undefined) {
      lines.push(`      Hint: ${r.hint}`);
    }
  }
  const counts = {
    ok: report.results.filter((r) => r.status === 'ok').length,
    fail: report.results.filter((r) => r.status === 'fail').length,
    skip: report.results.filter((r) => r.status === 'skip').length,
  };
  lines.push('');
  lines.push(
    `Result: ${counts.fail} failed, ${counts.ok} ok, ${counts.skip} skipped (total ${report.results.length})`,
  );
  lines.push(`Exit: ${report.exitCode}`);
  lines.push('');
  return lines.join('\n');
}
```

```typescript
import { getEncoding, type Tiktoken } from 'js-tiktoken';

let encoderCache: Tiktoken | null = null;

/**
 * lazy 初期化された cl100k_base エンコーダを返す。
 * cl100k_base は OpenAI 系 GPT-3.5/4 の BPE。Llama/Qwen 等の SentencePiece 系
 * モデルに対しては近似値（経験則として ±10〜25% の誤差）。
 * 安全マージン (TOKEN_SAFETY_MARGIN) を掛けて overflow を防ぐ前提で使う。
 */
function getEncoder(): Tiktoken {
  if (encoderCache === null) {
    encoderCache = getEncoding('cl100k_base');
  }
  return encoderCache;
}

export const TOKEN_SAFETY_MARGIN = 0.85;

export function effectiveBudget(maxTokens: number): number {
  if (maxTokens <= 0) return 0;
  return Math.floor(maxTokens * TOKEN_SAFETY_MARGIN);
}

export function countTokens(text: string): number {
  if (text.length === 0) return 0;
  try {
    return getEncoder().encode(text).length;
  } catch {
    // フォールバック: encoder 失敗時は 1 token ≒ 4 chars と仮定
    return Math.ceil(text.length / 4);
  }
}

export function truncateToTokens(text: string, budget: number): string {
  if (budget <= 0) return '';
  if (text.length === 0) return '';
  try {
    const encoder = getEncoder();
    const tokens = encoder.encode(text);
    if (tokens.length <= budget) return text;
    return encoder.decode(tokens.slice(0, budget));
  } catch {
    // フォールバック: 文字数ベース切り詰め (1 token ≒ 4 chars)
    const charBudget = budget * 4;
    return text.length <= charBudget ? text : text.slice(0, charBudget);
  }
}
```

```typescript
import { countTokens, truncateToTokens, effectiveBudget } from './tokenizer.js';
import type { NexusResult } from './types.js';

export interface TruncateInput {
  diff: string;
  contexts: NexusResult[];
  maxTokens: number;
}

export interface TruncateOutput {
  diff: string;
  contexts: NexusResult[];
}

function splitDiffBlocks(diff: string): string[] {
  const lines = diff.split('\n');
  const blocks: string[] = [];
  let current: string[] = [];
  for (const line of lines) {
    if (line.startsWith('diff --git ')) {
      if (current.length > 0) blocks.push(current.join('\n'));
      current = [line];
    } else {
      current.push(line);
    }
  }
  if (current.length > 0) blocks.push(current.join('\n'));
  return blocks;
}

function truncateDiffByTokens(diff: string, budget: number): string {
  if (budget <= 0) return '';
  if (countTokens(diff) <= budget) return diff;

  const blocks = splitDiffBlocks(diff);
  if (blocks.length === 0) return '';

  while (blocks.length > 1 && countTokens(blocks.join('\n')) > budget) {
    blocks.pop();
  }

  let joined = blocks.join('\n');
  if (countTokens(joined) <= budget) return joined;

  // 最終 block 内を末尾から token 単位で切る。先頭ヘッダは保持。
  const last = blocks[blocks.length - 1]!;
  const newlineIdx = last.indexOf('\n');
  const header = newlineIdx === -1 ? last : last.slice(0, newlineIdx + 1);
  const body = newlineIdx === -1 ? '' : last.slice(newlineIdx + 1);
  const headerTokens = countTokens(header);
  const remainingBudget = budget - headerTokens;
  const truncatedBody = remainingBudget > 0 ? truncateToTokens(body, remainingBudget) : '';
  blocks[blocks.length - 1] = header + truncatedBody;
  joined = blocks.join('\n');
  return joined;
}

function truncateContextsByTokens(contexts: NexusResult[], budget: number): NexusResult[] {
  if (contexts.length === 0) return [];
  let total = contexts.reduce((sum, c) => sum + countTokens(c.content), 0);
  if (total <= budget) return contexts;

  const remaining = [...contexts];
  while (total > budget && remaining.length > 0) {
    let longestIdx = 0;
    let longestTokens = countTokens(remaining[0]!.content);
    for (let i = 1; i < remaining.length; i++) {
      const t = countTokens(remaining[i]!.content);
      if (t > longestTokens) {
        longestIdx = i;
        longestTokens = t;
      }
    }
    remaining.splice(longestIdx, 1);
    total -= longestTokens;
  }
  return remaining;
}

export function build({ diff, contexts, maxTokens }: TruncateInput): TruncateOutput {
  const budget = effectiveBudget(maxTokens);
  const diffBudget = Math.floor(budget * 0.6);
  const contextBudget = budget - diffBudget;

  return {
    diff: truncateDiffByTokens(diff, diffBudget),
    contexts: truncateContextsByTokens(contexts, contextBudget),
  };
}
```

```typescript
import { readFile } from 'node:fs/promises';
import { join } from 'node:path';
import { execFile } from 'node:child_process';
import { promisify } from 'node:util';

const execFileAsync = promisify(execFile);

async function findGitRoot(cwd: string): Promise<string | null> {
  try {
    const { stdout } = await execFileAsync('git', ['rev-parse', '--show-toplevel'], { cwd });
    const trimmed = stdout.trim();
    return trimmed.length > 0 ? trimmed : null;
  } catch {
    return null;
  }
}

export interface PromptFileLookupResult {
  readonly path: string | null;
  readonly content: string | null;
}

export async function loadPromptFile(
  cwd: string = process.cwd(),
): Promise<PromptFileLookupResult> {
  const root = await findGitRoot(cwd);
  if (root === null) return { path: null, content: null };

  const candidate = join(root, '.github', 'nxc.prompt.md');
  try {
    const content = await readFile(candidate, 'utf8');
    return { path: candidate, content };
  } catch (err) {
    if ((err as NodeJS.ErrnoException).code === 'ENOENT') {
      return { path: candidate, content: null };
    }
    throw err;
  }
}

export async function findPromptFile(cwd: string = process.cwd()): Promise<string | null> {
  const result = await loadPromptFile(cwd);
  return result.content !== null ? result.path : null;
}
```

```typescript
function buildSystem(lang: Lang, customSuffix?: string): string {

  const base = [
    'あなたは熟練ソフトウェアエンジニアです。',
    '以下の git diff と関連コンテキストを読み、Conventional Commits v1.0.0 に厳格に準拠した',

    '- description は命令形で簡潔に',
    '- 本文が必要なら空行を1行挟んで記述',
```

```typescript
async function checkPromptFile(deps: DoctorDeps): Promise<CheckResult> {
  try {
    const path = await findPromptFile(deps.cwd ?? process.cwd());
    if (path === null) {
      return {
        title: 'Custom prompt file',
        status: 'skip',
        detail: 'no .github/nxc.prompt.md',
      };
    }
    return { title: 'Custom prompt file', status: 'ok', detail: path };
  } catch (err) {
    return {
      title: 'Custom prompt file',
      status: 'warn',
      detail: err instanceof Error ? err.message : String(err),
    };
  }
}
```
